package test01;

import java.io.File;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.OutputStreamWriter;

public class Core {
	public static void main(String[] args) throws IOException {
		if(args.length==2) 
		{
			tip2(args[0],args[1]);
		}
		else if(args.length==3)
		{
			tip3(args[0],args[1],args[2]);
		}
		else 
		{
			System.out.println("参数输入有误,请重新输入!");
		}
	}
	public static void tip2(String inputfile,String outputfile) throws IOException {
		ReadTxt5.readFileByLines(inputfile);
		ReadTxt5.province();
		ReadTxt5.change(ReadTxt5.province);
		ReadTxt5.rank(ReadTxt5.string);
		System.out.println("各省份及其所包含城市的范围：");
		ReadTxt5.area();
		File file = new File(outputfile);
		FileOutputStream out = new FileOutputStream(file);
		OutputStreamWriter wr =new OutputStreamWriter(out);		
		wr.write(ReadTxt5.location+"            "+ReadTxt5.province[0][1]+"\r\n\r\n");
		for(int m = 0;m < ReadTxt5.province.length;m++) 
		{
				for(int i=0;i<ReadTxt5.string.length;i++) 
				{
					if(ReadTxt5.string[i][0].equals(ReadTxt5.province[m][0]))
					{
						wr.write(ReadTxt5.string[i][1]+"            "+ReadTxt5.string[i][2]+"\r\n");	
					}
					else
					{
						continue;
					}
				}
				if(m!=ReadTxt5.province.length-1) 
				{
					ReadTxt5.location=ReadTxt5.province[m+1][0];
					wr.write(ReadTxt5.location+"            "+ReadTxt5.province[m+1][1]+"\r\n\r\n");
				}
				else
				break;
			}	
			wr.close();
	   }
	
	public static void tip3(String inputfile,String outputfile,String area) throws IOException  {
		ReadTxt5.readFileByLines(inputfile);
		ReadTxt5.province();
		ReadTxt5.change(ReadTxt5.province);
		ReadTxt5.rank(ReadTxt5.string);
		System.out.println("各省份及其所包含城市的范围：");
		ReadTxt5.area();
		File file = new File(outputfile);
		FileOutputStream out = new FileOutputStream(file);
		OutputStreamWriter wr =new OutputStreamWriter(out);
		ReadTxt5.location=area;
		wr.write(ReadTxt5.location+"            "+ReadTxt5.province[0][1]+"\r\n\r\n");
		for(int n =0 ;n < ReadTxt5.string.length;n++)
		{
			if(ReadTxt5.string[n][0].equals(ReadTxt5.location))
			{
				wr.write(ReadTxt5.string[n][1]+"            "+ReadTxt5.string[n][2]+"\r\n");
			}
			else
			{
				continue;
			}
		}
		wr.close();
	}
}
